   <script>
   (function(){
     var h=location.hostname, p=location.pathname;
     var enA=document.getElementById('link-en'), zhA=document.getElementById('link-zh');
     if(h.indexOf('readthedocs.io')!==-1){
       enA.href='https://backtrader.readthedocs.io/en/latest/';
       zhA.href='https://backtrader-zh.readthedocs.io/zh-cn/latest/';
     } else {
       var base='';
       var ei=p.indexOf('/en/'), zi=p.indexOf('/zh/');
       if(ei!==-1) base=p.substring(0,ei);
       else if(zi!==-1) base=p.substring(0,zi);
       enA.href=base+'/en/index.html';
       zhA.href=base+'/zh/index_zh.html';
     }
   })();
   </script>
